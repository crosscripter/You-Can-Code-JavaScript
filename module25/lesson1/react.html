Component State in React
const { useState } = React;

// Define the Counter component
function Counter() {

  // Define a state variable 'count' with an initial value of 0
  const [count, setCount] = useState(0);

  // Function to handle increment
  const handleIncrement = () => {
    setCount(count + 1); // Update the count state
  };

  return (
    <div>
      {/* Display the count */}
      <h2>Counter: {count}</h2> 

      {/* Button to increase count */} 
      <button onClick={handleIncrement}>Increase</button> 
    </div>
  );
}

ReactDOM.render(<Counter />, document.getElementById('root3'));
Passing Props in React
// Child component
function Display({ count }) {
  return (
    <div className="child">
      {/* Display the parent's count prop */}
      <h2>Counter: {count}</h2>
    </div>
  );
}

// Parent component
function Counter() {
  // State
  const [count, setCount] = useState(0);
  const handleIncrement = () => setCount(count + 1);

  return (
    <div className="parent">

      {/* Child component passing count prop */}
      <Display count={count} /> {/* Passing count as a prop */}

      <button onClick={handleIncrement}>Increase</button>
    </div>
  );
}

ReactDOM.render(<Counter />, document.getElementById('root4'));
